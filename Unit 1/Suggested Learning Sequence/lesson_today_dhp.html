Step 0 — Identify your sensor before connecting

🔍 Which sensor do you have?

Look at the label on your sensor module. Count the connector pins. This determines your wiring and the one line of code you need to change.

DHT11

Label says DHT11 · blue or white module · 3 pins

  • Temperature: ±2°C accuracy (integers only)
  • Humidity: ±5% accuracy
  • Minimum 1 s between readings
Digital → D2 port
Set: SENSOR = 'DHT11'

DHT22 / AM2302

Label says DHT22 or AM2302 · 3 pins

  • Temperature: ±0.5°C accuracy (float)
  • Humidity: ±2% accuracy
  • Minimum 2 s between readings
Digital → D2 port
Set: SENSOR = 'DHT22'

DHT20

Label says DHT20 · blue module · 4 pins

  • Temperature: ±0.5°C accuracy (float)
  • Humidity: ±3% accuracy
  • I²C address 0x38
I²C port (0x38)
Set: SENSOR = 'DHT20'

Not sure? Count the connector pins: 3 pins = DHT11 or DHT22 → digital D2 port. 4 pins = DHT20 → I²C port.

# main.py — Temperature + Humidity + PIR Rest Monitor
# FireBeetle 2 ESP32-E | EEDU IoT Kit | Lesson 2
# Builds on Lesson 1 (PIR) — adds temp & humidity sensor
#
# SENSORS:
#   PIR           → D3 port on IO Shield  (GPIO26)  ← from Lesson 1
#   DHT11 / DHT22 → D2 port on IO Shield  (GPIO25)  ← OR
#   DHT20         → I²C port on IO Shield (0x38)    ← choose below

# ── SET YOUR SENSOR ─────────────────────────────────────────────
# Change this to match your sensor label
SENSOR = 'DHT11'   # options: 'DHT11'  'DHT22'  'DHT20'
# ────────────────────────────────────────────────────────────────

from machine import Pin
import socket, time


# ════════════════════════════════════════════════════════════════
# SECTION 1 — Temp & Humidity Sensor
# read_sensor() always returns (temp_float, hum_float) or
# (None, None) regardless of which sensor type you chose above.
# ════════════════════════════════════════════════════════════════

if SENSOR in ('DHT11', 'DHT22'):
    # 3-pin Gravity cable → D2 port → GPIO25
    # Uses MicroPython's built-in dht library — no install needed
    import dht
    _d = dht.DHT11(Pin(25)) if SENSOR == 'DHT11' else dht.DHT22(Pin(25))

    def read_sensor():
        try:
            _d.measure()
            return float(_d.temperature()), float(_d.humidity())
        except OSError:
            return None, None

else:
    # DHT20 — 4-pin Gravity cable → I²C port → GPIO21 (SDA) / GPIO22 (SCL)
    # Uses I²C protocol at address 0x38
    # Inline driver — no external library needed
    from machine import I2C
    _i = I2C(0, scl=Pin(22), sda=Pin(21), freq=400000)
    _A = 0x38
    time.sleep_ms(100)
    try:
        s = _i.readfrom(_A, 1)[0]
        if (s & 0x18) != 0x18:
            for c in (b'\x1B\x00\x00', b'\x1C\x00\x00', b'\x1E\x00\x00'):
                _i.writeto(_A, c)
        print('DHT20 ready at 0x38')
    except Exception as e:
        print('DHT20 error:', e, '— check I2C port')

    def read_sensor():
        try:
            _i.writeto(_A, b'\xAC\x33\x00')
            time.sleep_ms(80)
            d = _i.readfrom(_A, 6)
            if d[0] & 0x80: return None, None
            h = round(((d[1]<<12)|(d[2]<<4)|(d[3]>>4))/1048576*100, 1)
            t = round(((d[3]&0x0F)<<16|d[4]<<8|d[5])/1048576*200-50, 1)
            return t, h
        except:
            return None, None

print('Sensor:', SENSOR, 'ready')


# ════════════════════════════════════════════════════════════════
# SECTION 2 — PIR Setup (unchanged from Lesson 1)
# ════════════════════════════════════════════════════════════════

pir = Pin(26, Pin.IN)              # D3 port = GPIO26
last_motion_ms = time.ticks_add(time.ticks_ms(), -61000)
CALM_SECONDS   = 60
WARMUP_MS      = 30000
startup_ms     = time.ticks_ms()

def check_pir():
    global last_motion_ms
    if pir.value() == 1:
        last_motion_ms = time.ticks_ms()
    return pir.value() == 1

def seconds_since_motion():
    return time.ticks_diff(time.ticks_ms(), last_motion_ms) // 1000


# ════════════════════════════════════════════════════════════════
# SECTION 3 — 3-Condition Rest Algorithm
#
# SELECTION evaluates all three conditions independently:
#   A) Motion   — room calm for ≥60 s
#   B) Temp     — 16.0°C to 22.0°C
#   C) Humidity — below 70%
#
# Score 0–3 → specific status message for Daniel
# ════════════════════════════════════════════════════════════════

TEMP_MIN = 16.0   # °C — sleep research: below this is too cold
TEMP_MAX = 22.0   # °C — above this disrupts sleep onset
HUM_MAX  = 70.0   # %  — above this feels stuffy

def evaluate_rest(temp, hum):
    """
    Evaluate 3 conditions and return status for Daniel.
    Returns (status, advice, score, motion_ok, temp_ok, hum_ok, secs)
    """
    motion_now = check_pir()
    secs       = seconds_since_motion()
    warmup     = time.ticks_diff(time.ticks_ms(), startup_ms) < WARMUP_MS

    # SELECTION — evaluate each condition as True or False
    motion_ok = (not motion_now) and (secs >= CALM_SECONDS) and (not warmup)
    temp_ok   = temp is not None and TEMP_MIN <= temp <= TEMP_MAX
    hum_ok    = hum  is not None and hum <= HUM_MAX

    score = (1 if motion_ok else 0) + (1 if temp_ok else 0) + (1 if hum_ok else 0)

    # Calm timer display
    mins   = secs // 60
    sec_r  = secs % 60
    t_str  = '{}m {}s'.format(mins, sec_r) if mins > 0 else '{}s'.format(secs)

    # SELECTION — pick the most useful message for Daniel
    if warmup:
        status = 'STARTING UP'
        advice = 'Sensors warming up — please wait 30 seconds.'

    elif score == 3:
        status = 'SUITABLE FOR REST'
        advice = 'All conditions met. Room is ready, Daniel.'

    elif score == 0:
        status = 'NOT SUITABLE'
        advice = 'Motion, temperature and humidity all need attention.'

    else:
        issues = []
        if not motion_ok:
            issues.append('motion {}.'.format('now' if motion_now else t_str + ' ago (need {}s calm)'.format(CALM_SECONDS)))
        if not temp_ok and temp is not None:
            direction = 'too cold' if temp < TEMP_MIN else 'too warm'
            issues.append('temp {:.1f}°C ({})'.format(temp, direction))
        if not hum_ok and hum is not None:
            issues.append('humidity {:.0f}% (above 70% threshold)'.format(hum))
        status = 'NEEDS ADJUSTMENT'
        advice = 'Fix: ' + ', '.join(issues) if issues else 'Sensor error — check connections.'

    return status, advice, score, motion_ok, temp_ok, hum_ok, secs


# ════════════════════════════════════════════════════════════════
# SECTION 4 — Build HTML Dashboard
# Blue-tinted dark design for Daniel:
#   - High contrast: #0a1628 bg / #bfdbfe text
#   - [OK] / [--] badges — accessible without colour alone
#   - Three tiles: Motion, Temperature, Humidity
#   - Progress bar and 3-condition score
#   - Auto-refresh every 5 seconds
# ════════════════════════════════════════════════════════════════

def build_page(temp, hum, status, advice, score, m_ok, t_ok, h_ok, secs):
    t_s = '{:.1f}°C'.format(temp) if temp is not None else '---'
    h_s = '{:.0f}%'.format(hum)  if hum  is not None else '---'

    syms  = {3:'\u2713', 2:'~', 1:'!', 0:'\u2717'}
    brds  = {3:'border:3px solid #93c5fd', 2:'border:3px dashed #60a5fa',
             1:'border:3px dotted #3b82f6', 0:'border:3px dotted #1d4ed8'}
    sym, brd = syms.get(score, '?'), brds.get(score, 'border:3px dotted #1d4ed8')

    ok_b, fail_b = '[OK]', '[--]'
    mb = ok_b if m_ok else fail_b
    tb = ok_b if t_ok else fail_b
    hb = ok_b if h_ok else fail_b

    filled = min(10, secs * 10 // CALM_SECONDS)
    bar    = '\u2588' * filled + '\u2591' * (10 - filled)

    return """<!DOCTYPE html>
<html lang="en"><head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width,initial-scale=1">
<meta http-equiv="refresh" content="5">
<title>Rest Monitor</title>
<style>
*{{box-sizing:border-box;margin:0;padding:0}}
body{{background:#0a1628;color:#bfdbfe;font-family:monospace;
     display:flex;flex-direction:column;align-items:center;
     min-height:100vh;padding:18px 12px;gap:12px}}
h1{{font-size:.72rem;letter-spacing:4px;color:#1e3a8a;text-transform:uppercase}}
.card{{background:#0f2044;border-radius:10px;
       padding:20px 22px;text-align:center;width:100%;max-width:420px}}
.sym{{font-size:3.8rem;line-height:1.1;color:#93c5fd}}
.status{{font-size:1.1rem;font-weight:bold;letter-spacing:1px;
         color:#bfdbfe;margin:7px 0 4px}}
.advice{{font-size:.8rem;color:#1e3a8a;line-height:1.55}}
.bwrap{{background:#0c1a38;border:1px solid #1e3a8a;border-radius:7px;
        padding:8px 10px;width:100%;max-width:420px}}
.bl{{font-size:.55rem;letter-spacing:2px;color:#1e3a8a;text-transform:uppercase}}
.bar{{font-size:1rem;letter-spacing:4px;color:#3b82f6;margin:4px 0 2px}}
.bs{{font-size:.6rem;color:#1a3566}}
.grid{{display:grid;grid-template-columns:repeat(3,1fr);gap:6px;
       width:100%;max-width:420px}}
.tile{{background:#0c1a38;border:1px solid #1a3566;
       border-radius:7px;padding:10px 6px;text-align:center}}
.tl{{font-size:.54rem;letter-spacing:1.5px;color:#1e3a8a;text-transform:uppercase}}
.tv{{font-size:.88rem;font-weight:bold;margin:4px 0 2px;color:#93c5fd}}
.ok{{color:#86efac}}.fail{{color:#f87171}}
.bdg{{font-size:.72rem;padding:1px 5px;border-radius:3px}}
.ok-bg{{background:#0a2e1a;color:#86efac}}.fail-bg{{background:#2e0a0a;color:#f87171}}
footer{{font-size:.6rem;color:#0f2044;text-align:center}}
</style></head>
<body>
  <h1>&#9632; REST ENVIRONMENT MONITOR</h1>
  <div class="card" style="{brd};background:#0f2044;border-radius:10px;
                            padding:20px 22px;text-align:center;width:100%;max-width:420px">
    <div class="sym">{sym}</div>
    <div class="status">{status}</div>
    <div class="advice">{advice}</div>
  </div>
  <div class="bwrap">
    <div class="bl">CALM TIMER</div>
    <div class="bar">{bar}</div>
    <div class="bs">{secs}s / {thresh}s threshold</div>
  </div>
  <div class="grid">
    <div class="tile">
      <div class="tl">Motion</div>
      <div class="tv"><span class="bdg {mc}">{mb}</span></div>
      <div style="font-size:.58rem;color:#1a3566">PIR · D3</div>
    </div>
    <div class="tile">
      <div class="tl">Temp</div>
      <div class="tv">{t}</div>
      <div><span class="bdg {tc}">{tb}</span></div>
    </div>
    <div class="tile">
      <div class="tl">Humidity</div>
      <div class="tv">{h}</div>
      <div><span class="bdg {hc}">{hb}</span></div>
    </div>
  </div>
  <footer>Updates every 5 s · no interaction needed · {snr}</footer>
</body></html>""".format(
        sym=sym, brd=brd, status=status, advice=advice,
        bar=bar, secs=min(secs,CALM_SECONDS), thresh=CALM_SECONDS,
        t=t_s, h=h_s, mb=mb, tb=tb, hb=hb,
        mc='ok-bg' if m_ok else 'fail-bg',
        tc='ok-bg' if t_ok else 'fail-bg',
        hc='ok-bg' if h_ok else 'fail-bg',
        snr=SENSOR
    )


# ════════════════════════════════════════════════════════════════
# SECTION 5 — Web Server (ITERATION)
# ════════════════════════════════════════════════════════════════

srv = socket.socket()
srv.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
srv.bind(('', 80)); srv.listen(3)
print('Server ready | Sensor: {} | PIR: GPIO26 | T/H: {}'.format(
    SENSOR, 'GPIO25 D2' if SENSOR != 'DHT20' else 'I2C 0x38'))

while True:                  # ITERATION
    try:
        conn, _ = srv.accept()
        conn.recv(1024)
        temp, hum = read_sensor()
        status, advice, score, m_ok, t_ok, h_ok, secs = evaluate_rest(temp, hum)
        print('[{}] T:{} H:{} Calm:{}s → {}'.format(SENSOR, temp, hum, secs, status))
        html = build_page(temp, hum, status, advice, score, m_ok, t_ok, h_ok, secs)
        conn.sendall(('HTTP/1.1 200 OK\r\nContent-Type: text/html\r\nConnection: close\r\n\r\n' + html).encode())
        conn.close()
        time.sleep(2)         # DHT22 needs 2 s between reads
    except Exception as e:
        print('Error:', e)
        try: conn.close()
        except: pass

🚨 Troubleshooting
  • Sensor reads None — wrong SENSOR = value? DHT11/22 must be on D2 port; DHT20 must be on I²C port
  • DHT20 init error — check the 4-pin cable is in an I²C port, not a D-port or A-port
  • Page not loading — same WiFi network as the board? IP typed correctly?
  • WiFi fails — 2.4 GHz network? Exact SSID/password in boot.py?
  • Saved to wrong place — choose "MicroPython device" not "This computer" in the save dialog
